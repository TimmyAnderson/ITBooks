void PrintIntArray(int MyArrayOfInt[], int ArrayLength)
{
	int		Counter=0;

	while(Counter<ArrayLength)
	{
		PrintString("Counter[");
		PrintInt(Counter);
		PrintString("]: ");
		PrintInt(MyArrayOfInt[Counter]);
		PrintString("! \n");

		Counter=Counter+1;
	}
}
