void PrintArray(int MyArrayOfInt[], int ArrayLength)
{
	int		Counter;

	while(Counter<ArrayLength)
	{
		PrintString("Counter[");
		PrintInt(Counter);
		PrintString("]: ");
		PrintInt(MyArrayOfInt[Counter]);
		PrintString("! \n");

		Counter=Counter+1;
	}
}
